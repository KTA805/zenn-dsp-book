## MFCCの計算手順

MFCCは以下の手順で計算されます。

1. STFTで周波数スペクトルを得る
2. メルフィルタバンクを適用する（メルスペクトログラム）
3. 対数を取る
4. DCT（離散コサイン変換）で圧縮する

```python
def compute_mfcc_manual(y, sr, n_mfcc=13):
    """MFCCを手順を追って計算する"""
    from scipy.signal import stft
    from scipy.fft import dct

    # Step 1: STFT
    _, _, Zxx = stft(y, fs=sr, nperseg=1024, noverlap=768)
    power = np.abs(Zxx) ** 2

    # Step 2: メルフィルタバンク
    n_mels = 26
    def hz_to_mel(hz): return 2595 * np.log10(1 + hz / 700)
    def mel_to_hz(mel): return 700 * (10 ** (mel / 2595) - 1)
    mel_pts = np.linspace(hz_to_mel(0), hz_to_mel(sr/2), n_mels + 2)
    hz_pts = mel_to_hz(mel_pts)
    bins = np.floor((1024 + 1) * hz_pts / sr).astype(int)
    fb = np.zeros((n_mels, 1024 // 2 + 1))
    for m in range(1, n_mels + 1):
        for k in range(bins[m-1], bins[m]):
            fb[m-1, k] = (k - bins[m-1]) / (bins[m] - bins[m-1] + 1e-10)
        for k in range(bins[m], bins[m+1]):
            fb[m-1, k] = (bins[m+1] - k) / (bins[m+1] - bins[m] + 1e-10)
    mel_spec = np.dot(fb, power)

    # Step 3: 対数
    log_mel = np.log(mel_spec + 1e-10)

    # Step 4: DCT
    mfcc = dct(log_mel, type=2, axis=0, norm='ortho')[:n_mfcc]
    return mfcc
```

## 2つの音声のMFCCを比較する

高い音と低い音でMFCCがどう違うか確認します。

```python
sr = 16000
t = np.linspace(0, 1, sr, endpoint=False)

# 440Hz（低め）と880Hz（高め）の音を生成
tone_low  = np.sin(2 * np.pi * 440 * t).astype(np.float32)
tone_high = np.sin(2 * np.pi * 880 * t).astype(np.float32)

# MFCCを計算
mfcc_low  = librosa.feature.mfcc(y=tone_low,  sr=sr, n_mfcc=13)
mfcc_high = librosa.feature.mfcc(y=tone_high, sr=sr, n_mfcc=13)

# 可視化
fig, axes = plt.subplots(1, 2, figsize=(14, 4))
for ax, mfcc, title in zip(axes,
    [mfcc_low, mfcc_high],
    ["MFCC (440Hz)", "MFCC (880Hz)"]):
    im = ax.imshow(mfcc, aspect='auto', origin='lower', cmap='RdBu_r')
    ax.set_title(title)
    ax.set_xlabel("Frame")
    ax.set_ylabel("MFCC coefficient")
    plt.colorbar(im, ax=ax)

plt.tight_layout()
plt.show()
```

色のパターンが異なります。周波数が違う音声は、MFCCでも異なるパターンになります。

## MFCCの各係数が何を表しているか

```python
# 時間平均のMFCCを比較
mean_low  = np.mean(mfcc_low, axis=1)
mean_high = np.mean(mfcc_high, axis=1)

plt.figure(figsize=(10, 4))
x = np.arange(13)
width = 0.35
plt.bar(x - width/2, mean_low,  width, label='440Hz', alpha=0.7)
plt.bar(x + width/2, mean_high, width, label='880Hz', alpha=0.7)
plt.xlabel("MFCC coefficient index")
plt.ylabel("Mean value")
plt.title("MFCCの各係数の平均値比較")
plt.xticks(x, [f"C{i}" for i in range(13)])
plt.legend()
plt.show()

# 差を数値で確認
diff = np.abs(mean_low - mean_high)
print("各係数の差の大きさ（大きいほど2つの音を区別しやすい）:")
for i, d in enumerate(diff):
    bar = "■" * int(d / max(diff) * 20)
    print(f"  C{i:2d}: {d:6.2f}  {bar}")
```

## 特徴量として使うための集約

MFCCは時間方向にフレームの列として得られます。分類モデルに入力するには固定長のベクトルにする必要があります。

```python
def mfcc_to_feature_vector(y, sr, n_mfcc=13):
    """MFCCを分類用の固定長ベクトルに変換する"""
    mfcc = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=n_mfcc)
    mean = np.mean(mfcc, axis=1)   # 時間平均
    std  = np.std(mfcc, axis=1)    # 時間標準偏差
    return np.concatenate([mean, std])  # 26次元ベクトル

# 使い方
feature = mfcc_to_feature_vector(tone_low, sr)
print(f"特徴量の次元数: {len(feature)}")   # 26
print(f"先頭6次元（mean）: {feature[:6].round(2)}")
print(f"次の6次元（std） : {feature[13:19].round(2)}")
```

`mean` だけ使う場合もありますが、`std` を加えると時間変化の情報も含まれるため、特に音節の区切りがある音声では精度が上がりやすいです。

## 異なる音声での比較

実際のデータに近い状況で確認します。

```python
# 3種類の音を生成（倍音付き）
def make_voice_like(freq, sr=16000, duration=1.0):
    t = np.linspace(0, duration, int(sr * duration), endpoint=False)
    y = (np.sin(2 * np.pi * freq * t) +
         0.5 * np.sin(2 * np.pi * freq * 2 * t) +
         0.3 * np.sin(2 * np.pi * freq * 3 * t))
    return (y / np.max(np.abs(y))).astype(np.float32)

sounds = {
    "低い声 (200Hz)": make_voice_like(200),
    "普通の声 (400Hz)": make_voice_like(400),
    "高い声 (800Hz)": make_voice_like(800),
}

features = {}
for name, y in sounds.items():
    features[name] = mfcc_to_feature_vector(y, sr=16000)

# 特徴量の差を確認
names = list(features.keys())
print("MFCCユークリッド距離（大きいほど異なる音声）:")
for i in range(len(names)):
    for j in range(i+1, len(names)):
        dist = np.linalg.norm(features[names[i]] - features[names[j]])
        print(f"  {names[i]} ↔ {names[j]}: {dist:.2f}")
```

異なる音声ほどMFCC特徴量の距離が大きくなります。これが音声分類の基礎です。

## まとめ

MFCCのポイントを整理します。

| 項目 | 内容 |
|------|------|
| 次元数 | 通常13〜40次元 |
| 時間集約 | mean + std で固定長ベクトルに |
| 低次係数 | スペクトルの全体的な形状（声の太さ） |
| 高次係数 | 細かい周波数変化（声のざらつきなど） |
